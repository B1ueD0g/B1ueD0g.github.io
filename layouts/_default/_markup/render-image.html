{{- $u := urls.Parse .Destination -}} {{- $src := $u.String -}} {{- $width := "" -}} {{- $height := "" -}} {{- if not $u.IsAbs -}} {{- $path := strings.TrimPrefix "./" $u.Path -}} {{- with or (.PageInner.Resources.Get $path) (resources.Get $path) -}} {{- $src = .RelPermalink -}} {{- $width = printf "%d" .Width -}} {{- $height = printf "%d" .Height -}} {{- with $u.RawQuery -}} {{- $src = printf "%s?%s" $src . -}} {{- end -}} {{- with $u.Fragment -}} {{- $src = printf "%s#%s" $src . -}} {{- end -}} {{- end -}} {{- end -}} {{- $attributes := merge .Attributes (dict "alt" .Text "src" $src "loading" "lazy" "decoding" "async") -}} {{- with .Title -}} {{- $attributes = merge $attributes (dict "title" (transform.HTMLEscape .)) -}} {{- end -}} {{- if and $width $height -}} {{- $attributes = merge $attributes (dict "width" $width "height" $height) -}} {{- end -}}
